
class Solution {
    public int maxPathSum(List<Integer> arr1, List<Integer> arr2) {
        int i=0;
        int j =0;
        int sum1 = 0;
        int sum2=0;
        int result=0;
        while(i<arr1.size() && j<arr2.size()){
            if(arr1.get(i)<arr2.get(j)){
                sum1+=arr1.get(i++);
            }
            else if(arr2.get(j)<arr1.get(i)){
                sum2+=arr2.get(j++);
            }
            else{
                result+=Math.max(sum1,sum2);
                result+=arr1.get(i);
                sum1=0;
                sum2=0;
                i++;
                j++;
            }
         
        }
        while(i<arr1.size()){
            sum1+=arr1.get(i);
            i++;
        }
        while(j<arr2.size()){
            sum2+=arr2.get(j);
            j++;
        }
        int remsum=Math.max(sum1,sum2);
        result+=remsum;
        return result;
        
    }
}
